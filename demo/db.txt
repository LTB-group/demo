CREATE DATABASE HospitalTechDB;
GO

USE HospitalTechDB;
GO

CREATE TABLE Roles (
    RoleId INT IDENTITY(1,1) PRIMARY KEY,
    RoleName NVARCHAR(50) NOT NULL
);

CREATE TABLE Users (
    UserId INT IDENTITY(1,1) PRIMARY KEY,
    FullName NVARCHAR(150) NOT NULL,
    Login NVARCHAR(100) NOT NULL UNIQUE,
    Password NVARCHAR(100) NOT NULL,
    RoleId INT NOT NULL FOREIGN KEY REFERENCES Roles(RoleId)
);

CREATE TABLE Manufacturers (
    ManufacturerId INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(100) NOT NULL
);

CREATE TABLE Categories (
    CategoryId INT IDENTITY(1,1) PRIMARY KEY,
    Name NVARCHAR(100) NOT NULL
);

CREATE TABLE Locations (
    LocationId INT IDENTITY(1,1) PRIMARY KEY,
    Address NVARCHAR(255) NOT NULL
);

CREATE TABLE Products (
    ProductCode NVARCHAR(20) PRIMARY KEY,
    Name NVARCHAR(150) NOT NULL,
    Price DECIMAL(18,2) NOT NULL,
    ManufacturerId INT NOT NULL FOREIGN KEY REFERENCES Manufacturers(ManufacturerId),
    CategoryId INT NOT NULL FOREIGN KEY REFERENCES Categories(CategoryId),
    DiscountAmount INT NOT NULL DEFAULT 0,
    LocationId INT NOT NULL FOREIGN KEY REFERENCES Locations(LocationId),
    Description NVARCHAR(MAX),
    PhotoPath NVARCHAR(255)
);
GO
